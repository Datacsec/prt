async function downloadFile(bank, partition_date, sequence, filename) {
  try {
    const params = new URLSearchParams({
      bank, partition_date, sequence, filename
    });

    const url = `${API_BASE_URL}/InssuranceFiles/downloadInssuranceResultsFile?${params}`;

    // 👇 Récupérer le token
    const token = localStorage.getItem('token')

    // 👇 Utiliser fetch avec le token dans le header
    const response = await fetch(url, {
      method: 'GET',
      headers: {
        'Authorization': `Bearer ${token}`
      }
    })

    if (!response.ok) {
      throw new Error(`Erreur HTTP: ${response.status}`)
    }

    // 👇 Convertir la réponse en blob pour le téléchargement
    const blob = await response.blob()
    const objectUrl = window.URL.createObjectURL(blob)

    const link = document.createElement('a')
    link.href = objectUrl
    link.setAttribute('download', filename)
    document.body.appendChild(link)
    link.click()
    link.remove()
    window.URL.revokeObjectURL(objectUrl)

  } catch (error) {
    console.error('Erreur telechargement :', error)
  }
}
