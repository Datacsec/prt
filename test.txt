public Map<String, Object> listFolderTree(String directoryPrefix) {
    String prefix = (directoryPrefix != null && !directoryPrefix.isBlank() && !directoryPrefix.endsWith("/"))
            ? directoryPrefix + "/"
            : directoryPrefix;

    ListObjectsV2Request req = ListObjectsV2Request.builder()
            .bucket(bucketName)
            .prefix(prefix)
            .maxKeys(1000)
            .build();

    ListObjectsV2Response res = s3.listObjectsV2(req);

    Map<String, Object> tree = new LinkedHashMap<>();

    for (S3Object obj : res.contents()) {
        String key = obj.key();

        // Supprimer le préfixe de base
        String relativePath = (prefix != null && key.startsWith(prefix))
                ? key.substring(prefix.length())
                : key;

        if (relativePath.isBlank()) continue;

        // Décomposer le chemin en segments
        String[] parts = relativePath.split("/");
        Map<String, Object> current = tree;

        for (int i = 0; i < parts.length; i++) {
            String part = parts[i];
            boolean isFile = (i == parts.length - 1) && !relativePath.endsWith("/");

            if (isFile) {
                current.put(part, "FILE"); // ou obj.key() pour le chemin complet
            } else {
                current.putIfAbsent(part, new LinkedHashMap<String, Object>());
                current = (Map<String, Object>) current.get(part);
            }
        }
    }

    return tree;
}
