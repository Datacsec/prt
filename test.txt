public FileNode listFolderTree(String directoryPrefix) {
    String prefix = (directoryPrefix != null && !directoryPrefix.isBlank() && !directoryPrefix.endsWith("/"))
            ? directoryPrefix + "/"
            : directoryPrefix;

    ListObjectsV2Request req = ListObjectsV2Request.builder()
            .bucket(bucketName)
            .prefix(prefix)
            .maxKeys(1000)
            .build();

    ListObjectsV2Response res = s3.listObjectsV2(req);

    // Noeud racine
    FileNode root = new FileNode("root", "Portail", "folder");
    Map<String, FileNode> folderMap = new LinkedHashMap<>();
    folderMap.put("", root);

    AtomicInteger fileCounter = new AtomicInteger(1);

    for (S3Object obj : res.contents()) {
        String key = obj.key();
        String relativePath = (prefix != null && key.startsWith(prefix))
                ? key.substring(prefix.length())
                : key;

        if (relativePath.isBlank()) continue;

        String[] parts = relativePath.split("/");
        StringBuilder pathBuilder = new StringBuilder();

        for (int i = 0; i < parts.length; i++) {
            String part = parts[i];
            String parentPath = pathBuilder.toString();
            if (i > 0) pathBuilder.append("/");
            pathBuilder.append(part);
            String currentPath = pathBuilder.toString();

            boolean isFile = (i == parts.length - 1) && !relativePath.endsWith("/");

            if (isFile) {
                String ext = part.contains(".") ? part.substring(part.lastIndexOf(".") + 1) : "";
                FileNode fileNode = new FileNode("f" + fileCounter.getAndIncrement(), part, "file");
                fileNode.setExtension(ext);
                fileNode.setSize(obj.size());
                folderMap.get(parentPath).getChildren().add(fileNode);
            } else {
                if (!folderMap.containsKey(currentPath)) {
                    FileNode folderNode = new FileNode(currentPath.replace("/", "-"), part, "folder");
                    folderMap.put(currentPath, folderNode);
                    folderMap.get(parentPath).getChildren().add(folderNode);
                }
            }
        }
    }

    return root;
}
