import json
from pathlib import Path

def charger_fichier_source(path: str) -> List[str]:
    with open(path) as f:
        return f.readlines()

def extraire_lignes(lignes: List[str], debut: int, fin: int) -> str:
    """Lignes 1-indexées."""
    return "".join(lignes[debut-1:fin])

def reconstituer_contexte(champ_dep: dict, builder: dict, fichier_source: str) -> str:
    """
    Reconstitue un bloc Scala contenant :
    - Les val nécessaires dans l'ordre
    - Les withColumn/alias qui concernent le champ
    """
    lignes = charger_fichier_source(fichier_source)
    
    parties = []
    parties.append(f"// ===== CHAMP : {champ_dep['champ']} (type: {champ_dep['type']}) =====\n")
    parties.append(f"// Dataframes sources : {', '.join(champ_dep['dataframes_sources'])}\n\n")
    
    # Les val impliquées dans l'ordre d'apparition dans le code
    vals_triees = sorted(
        [v for v in builder["vals"] if v["nom"] in champ_dep["vals_impliquees"]],
        key=lambda x: x["ligneDebut"]
    )
    
    for val_def in vals_triees:
        parties.append(f"// --- val {val_def['nom']} (ligne {val_def['ligneDebut']}) ---\n")
        parties.append(extraire_lignes(lignes, val_def["ligneDebut"], val_def["ligneFin"]))
        parties.append("\n\n")
    
    # Les sources directes du champ
    parties.append(f"// --- Construction directe du champ '{champ_dep['champ']}' ---\n")
    for s in champ_dep["sources_directes"]["via_alias"]:
        parties.append(f"// alias ligne {s['ligne']}\n")
        parties.append(s["codeExpression"] + f".as(\"{champ_dep['champ']}\")\n\n")
    
    for s in champ_dep["sources_directes"]["via_withColumn"]:
        parties.append(f"// withColumn ligne {s['ligne']}\n")
        parties.append(f".withColumn(\"{champ_dep['champ']}\", {s['codeExpression']})\n\n")
    
    return "".join(parties)

def main(fichier_builder: str):
    with open("output/dependencies.json") as f:
        deps = json.load(f)
    with open("output/index.json") as f:
        index = json.load(f)
    
    builder = index["builders"][deps["builder"]]
    
    Path("output/contexts").mkdir(parents=True, exist_ok=True)
    
    for champ_dep in deps["champs"]:
        if champ_dep["nb_val_dans_chaine"] == 0:
            continue
        
        contexte = reconstituer_contexte(champ_dep, builder, fichier_builder)
        
        with open(f"output/contexts/{champ_dep['champ']}.scala", "w") as f:
            f.write(contexte)
    
    print(f"✅ Contextes écrits dans output/contexts/")

if __name__ == "__main__":
    import sys
    main(sys.argv[1])  # chemin du builder
